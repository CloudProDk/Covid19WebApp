import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { Observable, of } from 'rxjs';
import { Movie } from './movie';
import { Genre } from 'src/app/models/genre';
import { NgxSpinnerService } from 'ngx-spinner';
import { SearchResult } from 'src/app/models/search-result';

@Injectable({
  providedIn: 'root',
})
export class MovieService {
  private url = 'https://api.themoviedb.org/3/';
  private apiKey = 'a489ee60e8b906cfa58b9dd0fa595d25';

  public searchResults: Movie[] = [];

  constructor(private http: HttpClient, private spinner: NgxSpinnerService) {}

  getActionMovies(): Observable<Movie> {
    let actionMoviesUrl = `${this.url}discover/movie?api_key=${this.apiKey}&with_genres=28`;
    return this.http.get<Movie>(actionMoviesUrl);
  }
  getComedyMovies(): Observable<Movie> {
    let comedyMoviesUrl = `${this.url}discover/movie?api_key=${this.apiKey}&with_genres=35`;
    return this.http.get<Movie>(comedyMoviesUrl);
  }
  getMovieGenres(): Observable<Genre> {
    let genresUrl = `${this.url}genre/movie/list?api_key=${this.apiKey}`;
    return this.http.get<Genre>(genresUrl);
  }
  getMoviesByGenreId(id: number) {
    let genreUrl = `${this.url}discover/movie?api_key=${this.apiKey}&with_genres=${id}`;
    return this.http.get<Movie>(genreUrl);
  }

  getActionMovie(id: number) {
    let movieUrl = `${this.url}movie/${id}?api_key=${this.apiKey}`;

    return this.http.get(movieUrl);
  }

  async getMovieById(id: number) {
    const movieUrl = `${this.url}movie/${id}?api_key=${this.apiKey}`;

    this.http.get<Movie>(movieUrl).subscribe((data) => {
      const movie: Movie = data;
      console.log(movie);
      return movie;
    });
  }

  async searchMovies(query: string) {
    this.spinner.show();
    const movieUrl = `${this.url}search/movie/?api_key=${this.apiKey}&language=en-US&query=${query}&page=1&include_adult=false`;
    await this.http
      .get<SearchResult>(movieUrl)
      .toPromise()
      .then((data) => {
        console.log(data.results);
        data.results.forEach((m) => {
          if (m.poster_path != null) {
            this.searchResults.push(m);
          }
        });
        this.spinner.hide();
        return;
      });
  }
}
