import { Component, OnInit, Input } from '@angular/core';
import { Movie } from 'src/app/services/movie/movie';
import { Trailer } from 'src/app/models/trailer';
import { NgxSpinnerService } from 'ngx-spinner';
import { HttpClient } from '@angular/common/http';
import { ToastrService } from 'ngx-toastr';

@Component({
  selector: 'app-youtube-player',
  templateUrl: './youtube-player.component.html',
  styleUrls: ['./youtube-player.component.scss']
})
export class YoutubePlayerComponent implements OnInit {

  @Input() movie: Movie;

  private url = 'https://api.themoviedb.org/3/';
  private apiKey = 'a489ee60e8b906cfa58b9dd0fa595d25';
  public movieTrailer: Trailer;

  constructor(private spinner: NgxSpinnerService, private http: HttpClient, private toastr: ToastrService) { }

  ngOnInit(): void {
    const tag = document.createElement('script');
    tag.src = 'https://www.youtube.com/iframe_api';
    document.body.appendChild(tag);

    this.getTrailer();
  }

  async getTrailer() {
    this.spinner.show();
    const movieUrl = `${this.url}movie/${this.movie.id}/videos?api_key=${this.apiKey}`;

    await this.http.get<Trailer>(movieUrl).subscribe(data => {
      if(data.results !== undefined){
        this.movieTrailer = data;
        console.log(this.movieTrailer.results[0]);
      } else {
        return;
      }
    });
    this.spinner.hide();
  }

}
