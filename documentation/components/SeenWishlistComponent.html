import { Component, OnInit } from '@angular/core';
import { Movie } from 'src/app/services/movie/movie';
import { WishlistService } from 'src/app/services/wishlist/wishlist.service';
import { MovieService } from 'src/app/services/movie/movie.service';
import { ToastrService } from 'ngx-toastr';
import { NgxSpinnerService } from 'ngx-spinner';
import { HttpClient } from '@angular/common/http';
import { Router } from '@angular/router';
import { environment } from 'src/environments/environment';
import { SelectedMovieComponent } from '../selected-movie/selected-movie.component';

@Component({
  selector: 'app-seen-wishlist',
  templateUrl: './seen-wishlist.component.html',
  styleUrls: ['./seen-wishlist.component.scss']
})
export class SeenWishlistComponent implements OnInit {

  private url = 'https://api.themoviedb.org/3/';
  private apiKey = 'a489ee60e8b906cfa58b9dd0fa595d25';
  wishList: Movie[] = [];



  // tslint:disable-next-line: max-line-length
  constructor(private wishListSvc: WishlistService, private movieSvc: MovieService, private toastr: ToastrService, private spinner: NgxSpinnerService, private http: HttpClient, private router: Router) { }

  ngOnInit() {
    this.loadWishList();
  }

  async loadWishList() {
    this.spinner.show()
    const t = await this.wishListSvc.getWishList(environment.USER_ID);
    await this.wishListSvc.WISHLIST.forEach(m => {
      if (m.wishlist_type === 'seen') {
        const movieUrl = `${this.url}movie/${m.movie_id}?api_key=${this.apiKey}`;
        console.log(m)
        this.http.get<Movie>(movieUrl).subscribe(data => {
          const movie: Movie = data;
          this.wishList.push(movie);
        });
      }
    });
    this.toastr.success('Wishlist loaded succesfully', 'Great success');
    this.spinner.hide();
  }

  public removeFromWishlist(movie: Movie) {
    console.log(movie);
  }

  navigate(movie: Movie) {
    SelectedMovieComponent.prototype.movie = movie;
    this.router.navigate(['wishlist/selected-movie']);
  }


}
