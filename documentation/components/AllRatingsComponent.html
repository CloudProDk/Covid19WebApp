import { Component, OnInit } from '@angular/core';
import { RatingService } from 'src/app/services/rating/rating.service';
import { ToastrService } from 'ngx-toastr';
import { NgxSpinnerService } from 'ngx-spinner';
import { Rating } from 'src/app/models/rating';
import { Movie } from 'src/app/services/movie/movie';
import { MovieService } from 'src/app/services/movie/movie.service';
import { environment } from 'src/environments/environment';
import { HttpClient } from '@angular/common/http';
import { SelectedSearchMovieComponent } from '../../../movies/child-components/selected-search-movie/selected-search-movie.component';
import { Router } from '@angular/router';

@Component({
  selector: 'app-all-ratings',
  templateUrl: './all-ratings.component.html',
  styleUrls: ['./all-ratings.component.scss']
})
export class AllRatingsComponent implements OnInit {

  private url = 'https://api.themoviedb.org/3/';
  private apiKey = 'a489ee60e8b906cfa58b9dd0fa595d25';
  // list: Rating[] = [];
  moviesRated: Movie[] = [];

  constructor(private ratingSvc: RatingService, private movieSvc: MovieService, private toastr: ToastrService, private spinner: NgxSpinnerService, private http: HttpClient, private router: Router) { }

  ngOnInit() {
    this.loadRatings();
  }

  async loadRatings() {
    console.log('got here')
    this.spinner.show();
    const t = await this.ratingSvc.getRatings(environment.USER_ID);
    await this.ratingSvc.RATINGS.forEach(m => {

      const movieUrl = `${this.url}movie/${m.movie_id}?api_key=${this.apiKey}`;
      console.log(m)
      this.http.get<Movie>(movieUrl).toPromise().then(data => {
        const movie: Movie = data;
        this.moviesRated.push(movie);
      });

    });
    this.toastr.success('Ratings loaded succesfully', 'Great success');
    this.spinner.hide();
  }

  loadMovie(movie: Movie) {
    SelectedSearchMovieComponent.prototype.movie = movie;
    this.router.navigate(['movies/selected']);
  }
}
