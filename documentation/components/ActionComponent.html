import { Component, OnInit } from '@angular/core';
import { MovieService } from '../../../../services/movie/movie.service';
import { WishlistService } from 'src/app/services/wishlist/wishlist.service';
import { Movie } from 'src/app/services/movie/movie';
import { WishList } from 'src/app/models/wishlist';
import { environment } from 'src/environments/environment';
import { SelectedSearchMovieComponent } from '../../child-components/selected-search-movie/selected-search-movie.component'
import { Router } from '@angular/router';

@Component({
  selector: 'app-action',
  templateUrl: './action.component.html',
  styleUrls: ['./action.component.scss'],
})
export class ActionComponent implements OnInit {
  results: {};
  constructor(private movieService: MovieService, private wishSvc: WishlistService, private router: Router) {}

  ngOnInit(): void {
    this.getActionMovies();
  }
  getActionMovies(): void {
    this.movieService.getActionMovies().subscribe((data) => {
      this.results = data.results;
    });
  }

  loadMovie(movie: Movie) {
    SelectedSearchMovieComponent.prototype.movie = movie;
    this.router.navigate(['movies/selected']);
  }
}
