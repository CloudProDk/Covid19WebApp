import { Component, OnInit, Input } from '@angular/core';
import { Rating } from 'src/app/models/rating';
import { environment } from 'src/environments/environment';
import { RatingService } from 'src/app/services/rating/rating.service';
import { Movie } from 'src/app/services/movie/movie';

@Component({
  selector: 'app-star-rating',
  templateUrl: './star-rating.component.html',
  styleUrls: ['./star-rating.component.scss']
})
export class StarRatingComponent implements OnInit {

  @Input() movie: Movie;
  currentRating: Rating = new Rating;

  constructor(private ratingSvc: RatingService) { }

  ngOnInit(): void {
    this.loadRating();
  }

  async loadRating() {
    await this.ratingSvc.getRatingForMovie(environment.USER_ID, this.movie.id).toPromise().then(data => {
      this.currentRating = data[0];
    });
  }

  counter() {
    let returnArray: number[] = [];
    for(let i = 1; i <= Math.round(this.movie.vote_average); i++) {
      returnArray.push(i);
    }
    return returnArray;
  }

  nonStar(){
    let returnArray: number[] = [];
    for(let i = 1; i <= (10 - Math.round(this.movie.vote_average)); i++) {
      returnArray.push(Math.round(this.movie.vote_average) + i);
    }
    return returnArray;
  }

  addRating(rating: number) {
    const r: Rating = {
      movie_id: this.movie.id,
      user_id: environment.USER_ID,
      rating: rating
    }
    if(this.currentRating) {
      this.ratingSvc.updateRating(r);
    } else {
      this.ratingSvc.addRating(r);
    }
    this.currentRating = r;
  }

  removeRating() {
    if(this.currentRating) {
      this.ratingSvc.deleteRating(this.currentRating);
      this.currentRating = new Rating;
    }
  }
}
